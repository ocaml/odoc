let migration_error loc missing_feature =
  Location.raise_errorf ~loc
    "migration error: %s is not supported before OCaml 5.02" missing_feature

let mk_ghost_attr name =
  {
    Ast_501.Parsetree.attr_name = { Location.txt = name; loc = Location.none };
    attr_payload = PStr [];
    attr_loc = Location.none;
  }

let rec concat_list_lit left right =
  let open Ast_502.Parsetree in
  let open Ast_502.Asttypes in
  match (left.pexp_desc, right.pexp_desc) with
  | _, Pexp_construct ({ txt = Lident "[]"; _ }, _) -> left
  | Pexp_construct ({ txt = Lident "[]"; _ }, _), _ -> right
  | ( Pexp_construct
        ( { txt = Lident "::"; loc },
          Some ({ pexp_desc = Pexp_tuple [ hd; tl ]; _ } as arg_expr) ),
      _ ) ->
      {
        left with
        pexp_desc =
          Pexp_construct
            ( { txt = Lident "::"; loc },
              Some
                {
                  arg_expr with
                  pexp_desc = Pexp_tuple [ hd; concat_list_lit tl right ];
                } );
      }
  | _ -> invalid_arg "Invalid ocaml.ppx.context's load_path"

let migrate_ppx_context_load_path expr =
  let open Ast_502.Parsetree in
  let loc = expr.pexp_loc in
  match expr.pexp_desc with
  | Pexp_tuple [ visible; hidden ] ->
      let migration_attr =
        {
          attr_name = { Location.txt = "ppxlib.migration.load_path"; loc };
          attr_loc = loc;
          attr_payload =
            PStr [ { pstr_loc = loc; pstr_desc = Pstr_eval (expr, []) } ];
        }
      in
      let expr' = concat_list_lit visible hidden in
      { expr' with pexp_attributes = migration_attr :: expr.pexp_attributes }
  | _ -> expr

let migrate_ppx_context_fields fields =
  List.map
    (fun (lident_loc, expr) ->
      match lident_loc.Ast_502.Asttypes.txt with
      | Longident.Lident "load_path" ->
          (lident_loc, migrate_ppx_context_load_path expr)
      | _ -> (lident_loc, expr))
    fields

let migrate_ppx_context_payload payload =
  let open Ast_502.Parsetree in
  match payload with
  | PStr
      [
        ({
           pstr_desc =
             Pstr_eval
               (({ pexp_desc = Pexp_record (fields, None) } as expr), attributes);
         } as stri);
      ] ->
      let new_fields = migrate_ppx_context_fields fields in
      let new_expr = { expr with pexp_desc = Pexp_record (new_fields, None) } in
      PStr [ { stri with pstr_desc = Pstr_eval (new_expr, attributes) } ]
  | _ -> payload

let rec copy_toplevel_phrase :
    Ast_502.Parsetree.toplevel_phrase -> Ast_501.Parsetree.toplevel_phrase =
  function
  | Ast_502.Parsetree.Ptop_def x0 ->
      Ast_501.Parsetree.Ptop_def (copy_structure x0)
  | Ast_502.Parsetree.Ptop_dir x0 ->
      Ast_501.Parsetree.Ptop_dir (copy_toplevel_directive x0)

and copy_toplevel_directive :
    Ast_502.Parsetree.toplevel_directive -> Ast_501.Parsetree.toplevel_directive
    =
 fun {
       Ast_502.Parsetree.pdir_name;
       Ast_502.Parsetree.pdir_arg;
       Ast_502.Parsetree.pdir_loc;
     } ->
  {
    Ast_501.Parsetree.pdir_name = copy_loc (fun x -> x) pdir_name;
    Ast_501.Parsetree.pdir_arg = Option.map copy_directive_argument pdir_arg;
    Ast_501.Parsetree.pdir_loc = copy_location pdir_loc;
  }

and copy_directive_argument :
    Ast_502.Parsetree.directive_argument -> Ast_501.Parsetree.directive_argument
    =
 fun { Ast_502.Parsetree.pdira_desc; Ast_502.Parsetree.pdira_loc } ->
  {
    Ast_501.Parsetree.pdira_desc = copy_directive_argument_desc pdira_desc;
    Ast_501.Parsetree.pdira_loc = copy_location pdira_loc;
  }

and copy_directive_argument_desc :
    Ast_502.Parsetree.directive_argument_desc ->
    Ast_501.Parsetree.directive_argument_desc = function
  | Ast_502.Parsetree.Pdir_string x0 -> Ast_501.Parsetree.Pdir_string x0
  | Ast_502.Parsetree.Pdir_int (x0, x1) ->
      Ast_501.Parsetree.Pdir_int (x0, Option.map (fun x -> x) x1)
  | Ast_502.Parsetree.Pdir_ident x0 ->
      Ast_501.Parsetree.Pdir_ident (copy_Longident_t x0)
  | Ast_502.Parsetree.Pdir_bool x0 -> Ast_501.Parsetree.Pdir_bool x0

and copy_expression :
    Ast_502.Parsetree.expression -> Ast_501.Parsetree.expression =
 fun {
       Ast_502.Parsetree.pexp_desc;
       Ast_502.Parsetree.pexp_loc;
       Ast_502.Parsetree.pexp_loc_stack;
       Ast_502.Parsetree.pexp_attributes;
     } ->
  {
    Ast_501.Parsetree.pexp_desc = copy_expression_desc pexp_desc;
    Ast_501.Parsetree.pexp_loc = copy_location pexp_loc;
    Ast_501.Parsetree.pexp_loc_stack = copy_location_stack pexp_loc_stack;
    Ast_501.Parsetree.pexp_attributes = copy_attributes pexp_attributes;
  }

and copy_expression_desc :
    Ast_502.Parsetree.expression_desc -> Ast_501.Parsetree.expression_desc =
  function
  | Ast_502.Parsetree.Pexp_ident x0 ->
      Ast_501.Parsetree.Pexp_ident (copy_loc copy_Longident_t x0)
  | Ast_502.Parsetree.Pexp_constant x0 ->
      Ast_501.Parsetree.Pexp_constant (copy_constant x0)
  | Ast_502.Parsetree.Pexp_let (x0, x1, x2) ->
      Ast_501.Parsetree.Pexp_let
        (copy_rec_flag x0, List.map copy_value_binding x1, copy_expression x2)
  | Ast_502.Parsetree.Pexp_function (params, tconstraint, body) ->
      let expr =
        match body with
        | Pfunction_body expr -> (
            match expr.pexp_desc with
            | Pexp_function _ ->
                (* We don't want this [fun] to be merged with the parent during
                   the round-trip. This attribute signals that this expression
                   really is the body of the function. *)
                let attr = mk_ghost_attr "ppxlib.migration.stop_taking" in
                let expr = copy_expression expr in
                { expr with pexp_attributes = attr :: expr.pexp_attributes }
            | _ -> copy_expression expr)
        | Pfunction_cases (cases, loc, attrs) ->
            {
              Ast_501.Parsetree.pexp_desc =
                Ast_501.Parsetree.Pexp_function (List.map copy_case cases);
              pexp_loc = copy_location loc;
              pexp_loc_stack = [];
              pexp_attributes = copy_attributes attrs;
            }
      in
      let expr =
        match tconstraint with
        | None -> expr
        | Some (Pconstraint c) ->
            {
              Ast_501.Parsetree.pexp_desc =
                Ast_501.Parsetree.Pexp_constraint (expr, copy_core_type c);
              pexp_loc = expr.pexp_loc;
              pexp_loc_stack = [];
              pexp_attributes = [];
            }
        | Some (Pcoerce (c1, c2)) ->
            let c1 = Option.map copy_core_type c1 in
            {
              Ast_501.Parsetree.pexp_desc =
                Ast_501.Parsetree.Pexp_coerce (expr, c1, copy_core_type c2);
              pexp_loc = expr.pexp_loc;
              pexp_loc_stack = [];
              pexp_attributes = [];
            }
      in
      let expr =
        List.fold_right
          (fun param expr ->
            match param with
            | {
             Ast_502.Parsetree.pparam_desc = Pparam_val (lbl, exp0, p);
             pparam_loc;
            } ->
                let pexp_desc =
                  Ast_501.Parsetree.Pexp_fun
                    ( copy_arg_label lbl,
                      Option.map copy_expression exp0,
                      copy_pattern p,
                      expr )
                in
                {
                  Ast_501.Parsetree.pexp_desc;
                  pexp_loc =
                    {
                      loc_start = param.pparam_loc.loc_start;
                      loc_end = expr.pexp_loc.loc_end;
                      loc_ghost = true;
                    };
                  pexp_loc_stack = [];
                  pexp_attributes = [];
                }
            | { pparam_desc = Pparam_newtype x; pparam_loc } ->
                let pexp_desc =
                  Ast_501.Parsetree.Pexp_newtype (copy_loc (fun x -> x) x, expr)
                in
                {
                  Ast_501.Parsetree.pexp_desc;
                  pexp_loc =
                    {
                      loc_start = param.pparam_loc.loc_start;
                      loc_end = expr.pexp_loc.loc_end;
                      loc_ghost = true;
                    };
                  pexp_loc_stack = [];
                  pexp_attributes = [];
                })
          params expr
      in
      expr.pexp_desc
      (* Ast_501.Parsetree.Pexp_function (List.map copy_case x0) *)
  | Ast_502.Parsetree.Pexp_apply (x0, x1) ->
      Ast_501.Parsetree.Pexp_apply
        ( copy_expression x0,
          List.map
            (fun x ->
              let x0, x1 = x in
              (copy_arg_label x0, copy_expression x1))
            x1 )
  | Ast_502.Parsetree.Pexp_match (x0, x1) ->
      Ast_501.Parsetree.Pexp_match (copy_expression x0, List.map copy_case x1)
  | Ast_502.Parsetree.Pexp_try (x0, x1) ->
      Ast_501.Parsetree.Pexp_try (copy_expression x0, List.map copy_case x1)
  | Ast_502.Parsetree.Pexp_tuple x0 ->
      Ast_501.Parsetree.Pexp_tuple (List.map copy_expression x0)
  | Ast_502.Parsetree.Pexp_construct (x0, x1) ->
      Ast_501.Parsetree.Pexp_construct
        (copy_loc copy_Longident_t x0, Option.map copy_expression x1)
  | Ast_502.Parsetree.Pexp_variant (x0, x1) ->
      Ast_501.Parsetree.Pexp_variant
        (copy_label x0, Option.map copy_expression x1)
  | Ast_502.Parsetree.Pexp_record (x0, x1) ->
      Ast_501.Parsetree.Pexp_record
        ( List.map
            (fun x ->
              let x0, x1 = x in
              (copy_loc copy_Longident_t x0, copy_expression x1))
            x0,
          Option.map copy_expression x1 )
  | Ast_502.Parsetree.Pexp_field (x0, x1) ->
      Ast_501.Parsetree.Pexp_field
        (copy_expression x0, copy_loc copy_Longident_t x1)
  | Ast_502.Parsetree.Pexp_setfield (x0, x1, x2) ->
      Ast_501.Parsetree.Pexp_setfield
        (copy_expression x0, copy_loc copy_Longident_t x1, copy_expression x2)
  | Ast_502.Parsetree.Pexp_array x0 ->
      Ast_501.Parsetree.Pexp_array (List.map copy_expression x0)
  | Ast_502.Parsetree.Pexp_ifthenelse (x0, x1, x2) ->
      Ast_501.Parsetree.Pexp_ifthenelse
        (copy_expression x0, copy_expression x1, Option.map copy_expression x2)
  | Ast_502.Parsetree.Pexp_sequence (x0, x1) ->
      Ast_501.Parsetree.Pexp_sequence (copy_expression x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_while (x0, x1) ->
      Ast_501.Parsetree.Pexp_while (copy_expression x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_for (x0, x1, x2, x3, x4) ->
      Ast_501.Parsetree.Pexp_for
        ( copy_pattern x0,
          copy_expression x1,
          copy_expression x2,
          copy_direction_flag x3,
          copy_expression x4 )
  | Ast_502.Parsetree.Pexp_constraint (x0, x1) ->
      Ast_501.Parsetree.Pexp_constraint (copy_expression x0, copy_core_type x1)
  | Ast_502.Parsetree.Pexp_coerce (x0, x1, x2) ->
      Ast_501.Parsetree.Pexp_coerce
        (copy_expression x0, Option.map copy_core_type x1, copy_core_type x2)
  | Ast_502.Parsetree.Pexp_send (x0, x1) ->
      Ast_501.Parsetree.Pexp_send (copy_expression x0, copy_loc copy_label x1)
  | Ast_502.Parsetree.Pexp_new x0 ->
      Ast_501.Parsetree.Pexp_new (copy_loc copy_Longident_t x0)
  | Ast_502.Parsetree.Pexp_setinstvar (x0, x1) ->
      Ast_501.Parsetree.Pexp_setinstvar
        (copy_loc copy_label x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_override x0 ->
      Ast_501.Parsetree.Pexp_override
        (List.map
           (fun x ->
             let x0, x1 = x in
             (copy_loc copy_label x0, copy_expression x1))
           x0)
  | Ast_502.Parsetree.Pexp_letmodule (x0, x1, x2) ->
      Ast_501.Parsetree.Pexp_letmodule
        ( copy_loc (fun x -> Option.map (fun x -> x) x) x0,
          copy_module_expr x1,
          copy_expression x2 )
  | Ast_502.Parsetree.Pexp_letexception (x0, x1) ->
      Ast_501.Parsetree.Pexp_letexception
        (copy_extension_constructor x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_assert x0 ->
      Ast_501.Parsetree.Pexp_assert (copy_expression x0)
  | Ast_502.Parsetree.Pexp_lazy x0 ->
      Ast_501.Parsetree.Pexp_lazy (copy_expression x0)
  | Ast_502.Parsetree.Pexp_poly (x0, x1) ->
      Ast_501.Parsetree.Pexp_poly
        (copy_expression x0, Option.map copy_core_type x1)
  | Ast_502.Parsetree.Pexp_object x0 ->
      Ast_501.Parsetree.Pexp_object (copy_class_structure x0)
  | Ast_502.Parsetree.Pexp_newtype (x0, x1) ->
      Ast_501.Parsetree.Pexp_newtype
        (copy_loc (fun x -> x) x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_pack x0 ->
      Ast_501.Parsetree.Pexp_pack (copy_module_expr x0)
  | Ast_502.Parsetree.Pexp_open (x0, x1) ->
      Ast_501.Parsetree.Pexp_open (copy_open_declaration x0, copy_expression x1)
  | Ast_502.Parsetree.Pexp_letop x0 ->
      Ast_501.Parsetree.Pexp_letop (copy_letop x0)
  | Ast_502.Parsetree.Pexp_extension x0 ->
      Ast_501.Parsetree.Pexp_extension (copy_extension x0)
  | Ast_502.Parsetree.Pexp_unreachable -> Ast_501.Parsetree.Pexp_unreachable

and copy_letop : Ast_502.Parsetree.letop -> Ast_501.Parsetree.letop =
 fun { Ast_502.Parsetree.let_; Ast_502.Parsetree.ands; Ast_502.Parsetree.body } ->
  {
    Ast_501.Parsetree.let_ = copy_binding_op let_;
    Ast_501.Parsetree.ands = List.map copy_binding_op ands;
    Ast_501.Parsetree.body = copy_expression body;
  }

and copy_binding_op :
    Ast_502.Parsetree.binding_op -> Ast_501.Parsetree.binding_op =
 fun {
       Ast_502.Parsetree.pbop_op;
       Ast_502.Parsetree.pbop_pat;
       Ast_502.Parsetree.pbop_exp;
       Ast_502.Parsetree.pbop_loc;
     } ->
  {
    Ast_501.Parsetree.pbop_op = copy_loc (fun x -> x) pbop_op;
    Ast_501.Parsetree.pbop_pat = copy_pattern pbop_pat;
    Ast_501.Parsetree.pbop_exp = copy_expression pbop_exp;
    Ast_501.Parsetree.pbop_loc = copy_location pbop_loc;
  }

and copy_direction_flag :
    Ast_502.Asttypes.direction_flag -> Ast_501.Asttypes.direction_flag =
  function
  | Ast_502.Asttypes.Upto -> Ast_501.Asttypes.Upto
  | Ast_502.Asttypes.Downto -> Ast_501.Asttypes.Downto

and copy_case : Ast_502.Parsetree.case -> Ast_501.Parsetree.case =
 fun {
       Ast_502.Parsetree.pc_lhs;
       Ast_502.Parsetree.pc_guard;
       Ast_502.Parsetree.pc_rhs;
     } ->
  {
    Ast_501.Parsetree.pc_lhs = copy_pattern pc_lhs;
    Ast_501.Parsetree.pc_guard = Option.map copy_expression pc_guard;
    Ast_501.Parsetree.pc_rhs = copy_expression pc_rhs;
  }

and copy_value_binding :
    Ast_502.Parsetree.value_binding -> Ast_501.Parsetree.value_binding =
 fun {
       Ast_502.Parsetree.pvb_pat;
       Ast_502.Parsetree.pvb_expr;
       Ast_502.Parsetree.pvb_constraint;
       Ast_502.Parsetree.pvb_attributes;
       Ast_502.Parsetree.pvb_loc;
     } ->
  {
    Ast_501.Parsetree.pvb_pat = copy_pattern pvb_pat;
    Ast_501.Parsetree.pvb_expr = copy_expression pvb_expr;
    Ast_501.Parsetree.pvb_constraint =
      Option.map copy_value_constraint pvb_constraint;
    Ast_501.Parsetree.pvb_attributes = copy_attributes pvb_attributes;
    Ast_501.Parsetree.pvb_loc = copy_location pvb_loc;
  }

and copy_pattern : Ast_502.Parsetree.pattern -> Ast_501.Parsetree.pattern =
 fun {
       Ast_502.Parsetree.ppat_desc;
       Ast_502.Parsetree.ppat_loc;
       Ast_502.Parsetree.ppat_loc_stack;
       Ast_502.Parsetree.ppat_attributes;
     } ->
  {
    Ast_501.Parsetree.ppat_desc = copy_pattern_desc ppat_desc;
    Ast_501.Parsetree.ppat_loc = copy_location ppat_loc;
    Ast_501.Parsetree.ppat_loc_stack = copy_location_stack ppat_loc_stack;
    Ast_501.Parsetree.ppat_attributes = copy_attributes ppat_attributes;
  }

and copy_pattern_desc :
    Ast_502.Parsetree.pattern_desc -> Ast_501.Parsetree.pattern_desc = function
  | Ast_502.Parsetree.Ppat_any -> Ast_501.Parsetree.Ppat_any
  | Ast_502.Parsetree.Ppat_var x0 ->
      Ast_501.Parsetree.Ppat_var (copy_loc (fun x -> x) x0)
  | Ast_502.Parsetree.Ppat_alias (x0, x1) ->
      Ast_501.Parsetree.Ppat_alias (copy_pattern x0, copy_loc (fun x -> x) x1)
  | Ast_502.Parsetree.Ppat_constant x0 ->
      Ast_501.Parsetree.Ppat_constant (copy_constant x0)
  | Ast_502.Parsetree.Ppat_interval (x0, x1) ->
      Ast_501.Parsetree.Ppat_interval (copy_constant x0, copy_constant x1)
  | Ast_502.Parsetree.Ppat_tuple x0 ->
      Ast_501.Parsetree.Ppat_tuple (List.map copy_pattern x0)
  | Ast_502.Parsetree.Ppat_construct (x0, x1) ->
      Ast_501.Parsetree.Ppat_construct
        ( copy_loc copy_Longident_t x0,
          Option.map
            (fun x ->
              let x0, x1 = x in
              (List.map (fun x -> copy_loc (fun x -> x) x) x0, copy_pattern x1))
            x1 )
  | Ast_502.Parsetree.Ppat_variant (x0, x1) ->
      Ast_501.Parsetree.Ppat_variant (copy_label x0, Option.map copy_pattern x1)
  | Ast_502.Parsetree.Ppat_record (x0, x1) ->
      Ast_501.Parsetree.Ppat_record
        ( List.map
            (fun x ->
              let x0, x1 = x in
              (copy_loc copy_Longident_t x0, copy_pattern x1))
            x0,
          copy_closed_flag x1 )
  | Ast_502.Parsetree.Ppat_array x0 ->
      Ast_501.Parsetree.Ppat_array (List.map copy_pattern x0)
  | Ast_502.Parsetree.Ppat_or (x0, x1) ->
      Ast_501.Parsetree.Ppat_or (copy_pattern x0, copy_pattern x1)
  | Ast_502.Parsetree.Ppat_constraint (x0, x1) ->
      Ast_501.Parsetree.Ppat_constraint (copy_pattern x0, copy_core_type x1)
  | Ast_502.Parsetree.Ppat_type x0 ->
      Ast_501.Parsetree.Ppat_type (copy_loc copy_Longident_t x0)
  | Ast_502.Parsetree.Ppat_lazy x0 ->
      Ast_501.Parsetree.Ppat_lazy (copy_pattern x0)
  | Ast_502.Parsetree.Ppat_unpack x0 ->
      Ast_501.Parsetree.Ppat_unpack
        (copy_loc (fun x -> Option.map (fun x -> x) x) x0)
  | Ast_502.Parsetree.Ppat_exception x0 ->
      Ast_501.Parsetree.Ppat_exception (copy_pattern x0)
  | Ast_502.Parsetree.Ppat_extension x0 ->
      Ast_501.Parsetree.Ppat_extension (copy_extension x0)
  | Ast_502.Parsetree.Ppat_open (x0, x1) ->
      Ast_501.Parsetree.Ppat_open (copy_loc copy_Longident_t x0, copy_pattern x1)

and copy_value_constraint :
    Ast_502.Parsetree.value_constraint -> Ast_501.Parsetree.value_constraint =
  function
  | Ast_502.Parsetree.Pvc_constraint { locally_abstract_univars; typ } ->
      Ast_501.Parsetree.Pvc_constraint
        {
          locally_abstract_univars =
            List.map (copy_loc (fun x -> x)) locally_abstract_univars;
          typ = copy_core_type typ;
        }
  | Ast_502.Parsetree.Pvc_coercion { ground; coercion } ->
      Ast_501.Parsetree.Pvc_coercion
        {
          ground = Option.map copy_core_type ground;
          coercion = copy_core_type coercion;
        }

and copy_core_type : Ast_502.Parsetree.core_type -> Ast_501.Parsetree.core_type
    =
 fun {
       Ast_502.Parsetree.ptyp_desc;
       Ast_502.Parsetree.ptyp_loc;
       Ast_502.Parsetree.ptyp_loc_stack;
       Ast_502.Parsetree.ptyp_attributes;
     } ->
  {
    Ast_501.Parsetree.ptyp_desc = copy_core_type_desc ptyp_loc ptyp_desc;
    Ast_501.Parsetree.ptyp_loc = copy_location ptyp_loc;
    Ast_501.Parsetree.ptyp_loc_stack = copy_location_stack ptyp_loc_stack;
    Ast_501.Parsetree.ptyp_attributes = copy_attributes ptyp_attributes;
  }

and copy_location_stack :
    Ast_502.Parsetree.location_stack -> Ast_501.Parsetree.location_stack =
 fun x -> List.map copy_location x

and copy_core_type_desc loc :
    Ast_502.Parsetree.core_type_desc -> Ast_501.Parsetree.core_type_desc =
  function
  | Ast_502.Parsetree.Ptyp_any -> Ast_501.Parsetree.Ptyp_any
  | Ast_502.Parsetree.Ptyp_var x0 -> Ast_501.Parsetree.Ptyp_var x0
  | Ast_502.Parsetree.Ptyp_arrow (x0, x1, x2) ->
      Ast_501.Parsetree.Ptyp_arrow
        (copy_arg_label x0, copy_core_type x1, copy_core_type x2)
  | Ast_502.Parsetree.Ptyp_tuple x0 ->
      Ast_501.Parsetree.Ptyp_tuple (List.map copy_core_type x0)
  | Ast_502.Parsetree.Ptyp_constr (x0, x1) ->
      Ast_501.Parsetree.Ptyp_constr
        (copy_loc copy_Longident_t x0, List.map copy_core_type x1)
  | Ast_502.Parsetree.Ptyp_object (x0, x1) ->
      Ast_501.Parsetree.Ptyp_object
        (List.map copy_object_field x0, copy_closed_flag x1)
  | Ast_502.Parsetree.Ptyp_class (x0, x1) ->
      Ast_501.Parsetree.Ptyp_class
        (copy_loc copy_Longident_t x0, List.map copy_core_type x1)
  | Ast_502.Parsetree.Ptyp_alias (x0, x1) ->
      Ast_501.Parsetree.Ptyp_alias (copy_core_type x0, x1.txt)
  | Ast_502.Parsetree.Ptyp_variant (x0, x1, x2) ->
      Ast_501.Parsetree.Ptyp_variant
        ( List.map copy_row_field x0,
          copy_closed_flag x1,
          Option.map (fun x -> List.map copy_label x) x2 )
  | Ast_502.Parsetree.Ptyp_poly (x0, x1) ->
      Ast_501.Parsetree.Ptyp_poly
        (List.map (fun x -> copy_loc (fun x -> x) x) x0, copy_core_type x1)
  | Ast_502.Parsetree.Ptyp_package x0 ->
      Ast_501.Parsetree.Ptyp_package (copy_package_type x0)
  | Ast_502.Parsetree.Ptyp_extension x0 ->
      Ast_501.Parsetree.Ptyp_extension (copy_extension x0)
  | Ast_502.Parsetree.Ptyp_open (x0, x1) ->
      migration_error loc "module open in types"

and copy_package_type :
    Ast_502.Parsetree.package_type -> Ast_501.Parsetree.package_type =
 fun x ->
  let x0, x1 = x in
  ( copy_loc copy_Longident_t x0,
    List.map
      (fun x ->
        let x0, x1 = x in
        (copy_loc copy_Longident_t x0, copy_core_type x1))
      x1 )

and copy_row_field : Ast_502.Parsetree.row_field -> Ast_501.Parsetree.row_field
    =
 fun {
       Ast_502.Parsetree.prf_desc;
       Ast_502.Parsetree.prf_loc;
       Ast_502.Parsetree.prf_attributes;
     } ->
  {
    Ast_501.Parsetree.prf_desc = copy_row_field_desc prf_desc;
    Ast_501.Parsetree.prf_loc = copy_location prf_loc;
    Ast_501.Parsetree.prf_attributes = copy_attributes prf_attributes;
  }

and copy_row_field_desc :
    Ast_502.Parsetree.row_field_desc -> Ast_501.Parsetree.row_field_desc =
  function
  | Ast_502.Parsetree.Rtag (x0, x1, x2) ->
      Ast_501.Parsetree.Rtag
        (copy_loc copy_label x0, x1, List.map copy_core_type x2)
  | Ast_502.Parsetree.Rinherit x0 ->
      Ast_501.Parsetree.Rinherit (copy_core_type x0)

and copy_object_field :
    Ast_502.Parsetree.object_field -> Ast_501.Parsetree.object_field =
 fun {
       Ast_502.Parsetree.pof_desc;
       Ast_502.Parsetree.pof_loc;
       Ast_502.Parsetree.pof_attributes;
     } ->
  {
    Ast_501.Parsetree.pof_desc = copy_object_field_desc pof_desc;
    Ast_501.Parsetree.pof_loc = copy_location pof_loc;
    Ast_501.Parsetree.pof_attributes = copy_attributes pof_attributes;
  }

and copy_attributes :
    Ast_502.Parsetree.attributes -> Ast_501.Parsetree.attributes =
 fun x -> List.map copy_attribute x

and copy_attribute : Ast_502.Parsetree.attribute -> Ast_501.Parsetree.attribute
    =
 fun {
       Ast_502.Parsetree.attr_name;
       Ast_502.Parsetree.attr_payload;
       Ast_502.Parsetree.attr_loc;
     } ->
  {
    Ast_501.Parsetree.attr_name = copy_loc (fun x -> x) attr_name;
    Ast_501.Parsetree.attr_payload =
      (match attr_name.txt with
      | "ocaml.ppx.context" ->
          copy_payload (migrate_ppx_context_payload attr_payload)
      | _ -> copy_payload attr_payload);
    Ast_501.Parsetree.attr_loc = copy_location attr_loc;
  }

and copy_payload : Ast_502.Parsetree.payload -> Ast_501.Parsetree.payload =
  function
  | Ast_502.Parsetree.PStr x0 -> Ast_501.Parsetree.PStr (copy_structure x0)
  | Ast_502.Parsetree.PSig x0 -> Ast_501.Parsetree.PSig (copy_signature x0)
  | Ast_502.Parsetree.PTyp x0 -> Ast_501.Parsetree.PTyp (copy_core_type x0)
  | Ast_502.Parsetree.PPat (x0, x1) ->
      Ast_501.Parsetree.PPat (copy_pattern x0, Option.map copy_expression x1)

and copy_structure : Ast_502.Parsetree.structure -> Ast_501.Parsetree.structure
    =
 fun x -> List.map copy_structure_item x

and copy_structure_item :
    Ast_502.Parsetree.structure_item -> Ast_501.Parsetree.structure_item =
 fun { Ast_502.Parsetree.pstr_desc; Ast_502.Parsetree.pstr_loc } ->
  {
    Ast_501.Parsetree.pstr_desc = copy_structure_item_desc pstr_desc;
    Ast_501.Parsetree.pstr_loc = copy_location pstr_loc;
  }

and copy_structure_item_desc :
    Ast_502.Parsetree.structure_item_desc ->
    Ast_501.Parsetree.structure_item_desc = function
  | Ast_502.Parsetree.Pstr_eval (x0, x1) ->
      Ast_501.Parsetree.Pstr_eval (copy_expression x0, copy_attributes x1)
  | Ast_502.Parsetree.Pstr_value (x0, x1) ->
      Ast_501.Parsetree.Pstr_value
        (copy_rec_flag x0, List.map copy_value_binding x1)
  | Ast_502.Parsetree.Pstr_primitive x0 ->
      Ast_501.Parsetree.Pstr_primitive (copy_value_description x0)
  | Ast_502.Parsetree.Pstr_type (x0, x1) ->
      Ast_501.Parsetree.Pstr_type
        (copy_rec_flag x0, List.map copy_type_declaration x1)
  | Ast_502.Parsetree.Pstr_typext x0 ->
      Ast_501.Parsetree.Pstr_typext (copy_type_extension x0)
  | Ast_502.Parsetree.Pstr_exception x0 ->
      Ast_501.Parsetree.Pstr_exception (copy_type_exception x0)
  | Ast_502.Parsetree.Pstr_module x0 ->
      Ast_501.Parsetree.Pstr_module (copy_module_binding x0)
  | Ast_502.Parsetree.Pstr_recmodule x0 ->
      Ast_501.Parsetree.Pstr_recmodule (List.map copy_module_binding x0)
  | Ast_502.Parsetree.Pstr_modtype x0 ->
